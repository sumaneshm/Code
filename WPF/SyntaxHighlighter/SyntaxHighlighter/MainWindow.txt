        private static readonly Brush _defaultColor = new SolidColorBrush(Colors.Black);
        private static readonly double _defaultSize = 12;

        public class Styler
        {
            public Brush ColorBrush { get; set; }
            public double Size { get; set; }

            public readonly static Styler Default = new Styler(_defaultColor, _defaultSize);

            public Styler(Brush brush, double size)
            {
                ColorBrush = brush;
                Size = size;
            }

            public Styler(string strInit)
            {

                ColorBrush = _defaultColor;
                Size = _defaultSize;

                var ar = strInit.Split(',');

                if (ar.Length == 1 && !String.IsNullOrWhiteSpace(ar[0]))
                {
                    ColorBrush = new SolidColorBrush((Color)ColorConverter.ConvertFromString(ar[0]));
                }

                if (ar.Length == 2 && !String.IsNullOrWhiteSpace(ar[1]))
                {
                    double d;
                    if (Double.TryParse(ar[1], out d))
                    {
                        Size = d;
                    }
                }

            }
        }



        private Dictionary<string, Styler> GetStylesDictionary(string[] colorCodings)
        {
            var styles = new Dictionary<string, Styler>();
            Array.ForEach(colorCodings, c =>
            {
                var sp = c.Split(':');
                if (sp.Length == 2 && !string.IsNullOrWhiteSpace(sp[0]) && !string.IsNullOrWhiteSpace(sp[1]))
                {
                    styles.Add(sp[0], new Styler(sp[1]));
                }
            });

            return styles;
        }

        private Styler GetStyle(string line, Dictionary<string, Styler> highlightings)
        {

            foreach (var oneKeyword in highlightings.Keys)
            {
                if (line.Contains(oneKeyword))
                {
                    return highlightings[oneKeyword];
                }
            }

            return Styler.Default;
        }

        private const string DefaultFontColor = "Black";
        private const double DefaultFontSize = 3d;
        private const string FontFormat = "<font color='{0}' size='{1}'>";
        private static readonly string DefaultFontFormat = string.Format(FontFormat, DefaultFontColor, DefaultFontSize);



        public Dictionary<string, string> GetLineFontStyles(string[] codings)
        {
            var allStyles = new Dictionary<string, string>();

            foreach (var code in codings)
            {
                var codeArray = code.Split(':');
                string fontColor = codeArray.Length > 1 ? codeArray[1] : DefaultFontColor;
                double size;
                double fontSize = codeArray.Length == 3 && double.TryParse(codeArray[2], out size) ? size : DefaultFontSize;
                allStyles.Add(codeArray[0], string.Format(FontFormat, fontColor, fontSize));
            }

            return allStyles;
        }

        private string GetFontStyle(string line, Dictionary<string, string> allStyles)
        {
            return (from oneKeyword in allStyles.Keys where line.Contains(oneKeyword) select allStyles[oneKeyword]).FirstOrDefault();
        }
